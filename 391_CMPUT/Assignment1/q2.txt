#include <stdio.h>
#include <sqlite3.h>

int main(int argc, char **argv) {

	sqlite3 *db;
	sqlite3_stmt *stmt;

	if (argc != 1) {
	  fprintf(stderr, "Usage: %s (Query and database hardcoded for assignment)\n", argv[0]);
	  return(1);
	} 

	int rc; //return code

  	rc = sqlite3_open("assignment1.db", &db);

  	if( rc ){
    	  fprintf(stderr, "Can't open database: %s\n", sqlite3_errmsg(db));
    	  sqlite3_close(db);
	  return(1);
  	}

	char *sql_stmt = "select airlines.airline_id from airlines where active = 'Y' and callsign is not null and iata is not null and icao is not null and country is not null";

  	rc = sqlite3_prepare_v2(db, sql_stmt, -1, &stmt, 0);

   	if (rc != SQLITE_OK) {  
          fprintf(stderr, "Preparation failed: %s\n", sqlite3_errmsg(db));
          sqlite3_close(db);
          return 1;
    	}    
      
    while((rc = sqlite3_step(stmt)) == SQLITE_ROW) {
        int col;
        for(col=0; col<sqlite3_column_count(stmt)-1; col++) {
          printf("%s|", sqlite3_column_text(stmt, col));
        }
        printf("%s", sqlite3_column_text(stmt, col));
        printf("\n");
    }

    sqlite3_finalize(stmt);

}
